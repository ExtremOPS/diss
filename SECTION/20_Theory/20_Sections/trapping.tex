non-magnetic approximation

Fresnel equations specify the relation of the incident, reflected, and 
transmitted electrical field of an electromagnetic wave


\cite[Chapter 7]{Jackson2013}

\begin{subequations}
\begin{align}
  \vb{E}_{\mathrm{r}} & =\fresnelr_{i}\,\vb{E}_{\mathrm{i}} \\[3mm]
  \vb{E}_{\mathrm{t}} & =\fresnelt_{i}\,\vb{E}_{\mathrm{i}}
\end{align}
\end{subequations}

\cite[Chapter 7]{Jackson2013}

\begin{subequations}
\begin{align}
  \fresnelr_{\mathrm{s}} & =
  \frac{2\,\nf\,\cos\incident}{\nf\,\cos\incident + \ns\,\cos\reflected} 
  \label{eq:th-fresnel-rs}\\[3mm]
  \fresnelt_{\mathrm{s}} & = \frac{\nf\,\cos\incident - 
  \ns\,\cos\reflected}{\nf\,\cos\incident + \ns\,\cos\reflected} 
  \label{eq:th-fresnel-ts}\\[3mm]
  \fresnelr_{\mathrm{p}} & =
  \frac{2\,\nf\,\cos\incident}{\ns\,\cos\incident + \nf\,\cos\reflected} 
  \label{eq:th-fresnel-rp}\\[3mm]
  \fresnelt_{\mathrm{p}} & = \frac{\ns\,\cos\incident - 
  \nf\,\cos\reflected}{\ns\,\cos\incident + \nf\,\cos\reflected}
\label{eq:th-fresnel-tp}
\end{align}
\end{subequations}

power of electromagnetic wave

\cite[Chapter 7]{Jackson2013}


for plane emw in free space with $\mu_{\mathrm{r}} = 1$ (non-magnetic)

\begin{equation}
  \intensity = \timeaverage{\abs{\vb{S}}} = 
  \timeaverage{\abs{\vb{E}_{0}\times\vb{H}_{0}}} = 
  \frac{1}{2}\,\frac{\refraction}{Z_{0}}\,\abs{\vb{E}_{0}}^{2} = 
  \frac{1}{2}\,\refraction\,c\epsilon_{0}\,\abs{\vb{E}_{0}}^{2}
\end{equation}

\begin{equation}
  P = \intensity\,A \propto \refraction\,\abs{\vb{E}_{0}}^{2}\,A
\end{equation}

\begin{equation}
  P_{\mathrm{i}} = P_{\mathrm{r}} + P_{\mathrm{t}}
\end{equation}


\begin{equation}
  \fresnelR_{i} = \frac{P_{\mathrm{r}}}{P_{\mathrm{i}}} = 
  \frac{\intensity_{\mathrm{r}}\,\nf\,w_{\mathrm{r}}}{\intensity_{\mathrm{i}}\,\nf\,w_{\mathrm{i}}} 
  =\frac{\abs{\fresnelr_{i}\,\vb{E}_{0}}^{2}}{\abs{\vb{E}_{0}}^{2}} = 
  \fresnelr_{i}^{2}
\end{equation}

\begin{equation}
  \fresnelT_{i} = \frac{P_{\mathrm{t}}}{P_{\mathrm{i}}} = 
  \frac{\intensity_{\mathrm{t}}\,\ns\,w_{\mathrm{t}}}{\intensity_{\mathrm{i}}\,\nf\,w_{\mathrm{i}}} 
  =\frac{\abs{\fresnelt_{i}\,\vb{E}_{0}}^{2}}{\abs{\vb{E}_{0}}^{2}}\,\frac{\ns\,w_{\mathrm{t}}}{\nf\,w_{\mathrm{i}}} 
  = \fresnelt_{i}^{2}\,\frac{\ns\,\cos\transmitted}{\nf\,\cos\incident}
\end{equation}
